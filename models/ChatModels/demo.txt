from dotenv import load_dotenv
import os
from langchain_groq import ChatGroq
# from langchain_google_genai import ChatGoogleGenAI
# from langchain_mistralai import ChatMistralAI

load_dotenv()  # Load environment variables from .env file

# This is for Groq LLM
GROQ_API_KEY = os.getenv("GROQ_API_KEY") # load the api key from environment variable

groq_llm = ChatGroq(api_key=GROQ_API_KEY,
    model = "llama-3.1-8b-instant")
response  = groq_llm.invoke("What are the tactics for quiting porn addiction? List them in a sequence also explain in detail also tell me the most effective one")
print(response.content)
